-- init(compensator.TransferFunction,0)
ASSERT TRUE FATAL "The order of numerator must be lower than the order of denominator"
ASSERT compensator.q[1] != 0 FATAL "Highest order coefficient in denominator cannot be zero"
_vi1 = 2
WHILE ( _vi1 <= compensator.n_den ) compensator.z[_vi1] = 0 _vi1 += 1 END WHILE
IF FALSE THEN compensator.j = 1 WHILE ( compensator.p[4 - compensator.j] == 0 AND compensator.j < 4 ) compensator.j = compensator.j + 1 END WHILE compensator.z[compensator.j] = compensator.output_o / compensator.p[4 - compensator.j] ELSE compensator.j = 1 WHILE ( compensator.p[4 - compensator.j] == 0 AND compensator.q[4 - compensator.j] == 0 ) compensator.j = compensator.j + 1 END WHILE compensator.z[compensator.j] = (compensator.output_o * compensator.q[1] - compensator.p[1] * compensator.s_in.signal[1]) / (compensator.p[4 - compensator.j] * compensator.q[1] - compensator.q[4 - compensator.j] * compensator.p[1]) END IF
